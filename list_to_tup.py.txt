
#converting list to set 

list1 = [1,2,3,4]
print(type(list1))
set1 = set(list1)
print(type(set1))

#converting tuple to set 

tup1 = (1,2,3,4,5) 
print(type(tup1)) 
set2 = set(tup1)
print(type(set2))

#converting list to tuple 

list2 = [1,2,3,4]
print(type(list2))
tup2 = tuple(list2)
print(type(tup2))
