'''
Set methods
'''

#1. add()
'''
set1 = {1,2,3,4}
print(set1)
set1.add(5)
print(set1)
'''

#2. update()
'''
set1 = {1,2,3,4}
set2 = {7,8,9}
print(set1)
print(set2)
set1.update(set2)
print(set1)
'''

#3. discard()
'''
set1 = {1,2,3,4,5}
print(set1)
set1.discard(4)
print(set1)
'''

#4. remove()
'''
set1 = {1,2,3,4,5,6}
print(set1)
set1.remove(2)
print(set1)
'''
